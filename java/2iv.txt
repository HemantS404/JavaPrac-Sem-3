class Main{
    int i;
    char c;
    float f;
    double d;
    String s;
    public static void main(String args[]){
        Main m = new Main();
        System.out.println("Default value of int is "+m.i);
        System.out.println("Default value of char is "+m.c);
        System.out.println("Default value of float is "+m.f);
        System.out.println("Default value of double is "+m.d);
        System.out.println("Default value of String is "+m.s);
    }
}
